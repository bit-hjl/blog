线段树

bit-hjl 2019-09-10 数据结构

介绍

一种区间修改只有log(n)复杂度的数据结构

# 一、线段树建树

线段树的作用较多，可以实现树状数组实现的各种功能，同时能维护其不能维护的数据，但前提是该组数据的运算满足结合律。

线段树的实现较树状数组明显较为复杂，他的修改和查询依旧是log(n)，但是空间复杂度提高了，一般防止越界开4倍空间。

线段树是一颗二叉树，通过不断二分区间来实现数据的分块化管理。线段树主要有以下几个函数实现，建树，单点修改，区间修改，查询，以及为了优化区间修改和查询的下推lazy标记的函数

建树：int lazy[400000]; int tree[400000];int num[100000];这三个数组是准备工作。l，r分别代表了当前操作区间的维护范围。如果l = r，证明已经找到了二叉树的叶子节点，将num中储存的数据代入，回溯。若不等，则继续向下查找。这里我们将二叉树的左子树规定为[l,mid],右子树规定为[mid + 1,r],然后递归修改，同时在结束当前子树的修改之后，将他们的父亲节点进行修改

void build(int l,int r,int rt)//建树
{
    if(l == r)
    {
        tree[rt] = num[l];
        return;
    }
    int mid = (l + r)>>1;
    build(l,mid,rt<<1);
    build(mid + 1,r,rt<<1|1);// |运算，无条件赋值，将二进制数的最后一位强制变为1
    tree[rt] = tree[rt<<1] + tree[rt<<1|1];//回溯上一个节点
}

# 二、线段树单点修改

如果仅仅是单点修改查询时不需要pushdown函数，如果混合查询却不使用该函数，显然由于在前边区间懒惰而没有下传的数据会因为单点修改后的回溯而覆盖掉，即之前的区间查询在当前区间的修改被抹除。单点修改类似于建树操作，只是在建树时递归修改左右子树的操作改为判断该点在哪个子树上，显然若n点大于mid在右边否则在左边，修改结束后回溯顺便修改父亲节点。

void changeone(int l,int r,int rt,int n,int x)//单点修改函数
{
    if(l == r)
    {
        tree[rt] += x;
        return;
    }
    int mid = (l + r) >>1;
    pushdown(rt,l,r);//若单点修改和区间修改同时进行时需要，否则不需要
    if(n <= mid)//寻找修改点的位置
        changeone(l,mid,rt<<1,n,x);
    else
        changeone(mid + 1,r,rt<<1|1,n,x);
    tree[rt] = tree[rt<<1] + tree[rt<<1|1];
}

# 三、线段树区间修改

显然，如果当前tree数组所维护的区间在修改范围内，则需要全部修改，同时挂上lazy标记，即只更新到了当前层和其上边的（上边的通过回溯实现），之后的层没有更新。若不是全部处于，则将该树向下传递，同时不能忘了传递该层 未下传的lazy标记和其中的数据（pushdown实现），判断当前L，R和mid的关系，进行传递

void changelist(int L,int R,int l,int r,int x,int rt)//区间修改，其中L,R为修改区间的范围，l，r为当前tree[rt]数组维护的数据范围
{
    if(L <= l && R >= r)
    {
        tree[rt] += (r - l + 1) * x;
        lazy[rt] += x;
        return;
    }
    int mid = (l + r)>>1;
    pushdown(rt,l,r);
    if(L <= mid)
        changelist( L, R, l,mid, x,rt*2);
    if(R > mid)
        changelist( L, R, mid + 1, r, x,rt*2 + 1);
    tree[rt] = tree[rt<<1] + tree[rt<<1|1];
}

# 四、线段树区间查询

类似于区间查询操作。其中ans即为每层的答案，最上层的即为答案

int search(int L,int R,int l,int r,int rt)//区间查询
{
    if(L <= l && R >= r)
        return tree[rt];
    int mid = (l + r)>>1;
    pushdown(rt,l,r);
    int ans = 0;
    if(L <= mid)
        ans += search(L,R,l,mid,rt<<1);
    if(R > mid)
        ans += search(L,R,mid + 1,r,rt<<1|1);
    return ans;
}

向下传递函数：很显然，如果我们在区间查询的过程中每次都将每一个数据修改一遍，那么复杂度是nlog(n)（单个点修改的复杂度是log(n)），将比普通算法更加费时（~~那还要这玩意何用）~~所以这时候就需要lazy数组的作用了，lazy数组表示当前数组每一个数修改了lazy[rt]的值，但是仅仅修改到了本层及以上的数据，以下的子树均没有动，类似一个缓存的作用。当真正需要子树的数据了，在向下传递。这样一来pushdown函数要做的事情就十分明显了，将左右子树分别加上上层lazy数组乘以他们的长度修改本层数据，将本层的lazy数据对应更新，同时删除上层的lazy数据即可

void pushdown(int rt,int l,int r)//将懒惰标记下传
{
    if(lazy[rt] != 0)
    {
        int mid = (l + r) / 2;
        tree[rt<<1] += lazy[rt]*(mid - l + 1);
        tree[rt<<1|1] += lazy[rt]*(r - mid);
        lazy[rt<<1] += lazy[rt];
        lazy[rt<<1|1] += lazy[rt];
        lazy[rt] = 0;//清空懒惰标记
    }
}

# 五、完整测试代码

#include<stdio.h>
#include<iostream>
using namespace std;
int lazy[400000];
int tree[400000];
int num[100000];
void pushdown(int rt, int l, int r)//将懒惰标记下传
{
    if (lazy[rt] != 0) {
        int mid = (l + r) / 2;
        tree[rt << 1] += lazy[rt] * (mid - l + 1);
        tree[rt << 1 | 1] += lazy[rt] * (r - mid);
        lazy[rt << 1] += lazy[rt];
        lazy[rt << 1 | 1] += lazy[rt];
        lazy[rt] = 0;//清空懒惰标记
    }
}
void build(int l, int r, int rt)//建树
{
    if (l == r) {
        tree[rt] = num[l];
        return;
    }
    int mid = (l + r) >> 1;
    build(l, mid, rt << 1);
    build(mid + 1, r, rt << 1 | 1);// |运算，无条件赋值，将二进制数的最后一位强制变为1
    tree[rt] = tree[rt << 1] + tree[rt << 1 | 1];//回溯上一个节点
}
void changeone(int l, int r, int rt, int n, int x)//单点修改函数
{
    if (l == r) {
        tree[rt] += x;
        return;
    }
    int mid = (l + r) >> 1;
    //pushdown(rt,l,r);//若单点修改和区间修改同时进行时需要，否则不需要
    if (n <= mid)//寻找修改点的位置
        changeone(l, mid, rt << 1, n, x);
    else
        changeone(mid + 1, r, rt << 1 | 1, n, x);
    tree[rt] = tree[rt << 1] + tree[rt << 1 | 1];
}

void changelist(int L, int R, int l, int r, int x, int rt)//区间修改，其中L,R为修改区间的范围，l，r为当前tree[rt]数组维护的数据范围
{
    if (L <= l && R >= r) {
        tree[rt] += (r - l + 1) * x;
        lazy[rt] += x;
        return;
    }
    int mid = (l + r) >> 1;
    pushdown(rt, l, r);
    if (L <= mid)
        changelist(L, R, l, mid, x, rt * 2);
    if (R > mid)
        changelist(L, R, mid + 1, r, x, rt * 2 + 1);
    tree[rt] = tree[rt << 1] + tree[rt << 1 | 1];
}
int search(int L, int R, int l, int r, int rt)//区间查询
{
    if (L <= l && R >= r)
        return tree[rt];
    int mid = (l + r) >> 1;
    pushdown(rt, l, r);
    int ans = 0;
    if (L <= mid)
        ans += search(L, R, l, mid, rt << 1);
    if (R > mid)
        ans += search(L, R, mid + 1, r, rt << 1 | 1);
    return ans;
}

int main() {
    int n;
    scanf("%d", &n);
    for (int i = 1; i <= n; i++)
        scanf("%d", &num[i]);
    build(1, n, 1);
    int a, b;
    int x, y, z;
    while (1) {
        char str;
        cin >> str;
        getchar();
        if (str == 'A')//输入为A时进行区间修改
        {
            scanf("%d%d%d", &x, &y, &z);//x，y为修改位置（闭区间），z为增加的数
            changelist(x, y, 1, n, z, 1);
        } else//单点修改
        {
            scanf("%d%d", &x, &y);//x为修改位置，y为该点要增加的数
            changeone(1, n, 1, x, y);
        }
        scanf("%d%d", &a, &b);//查询区间
        printf("%d", search(a, b, 1, n, 1));
    }
}
